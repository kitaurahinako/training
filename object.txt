Q2
    class Book

      attr_accessor: :title, :price

      def initialize(title)
        @title = title
      end

      def initialize(price)
        @price = price
      end

      def rename_title(title)
        @title = title
      end

      def price_increase(price)
        @price*2
      end

    end

    class Comic < Book

      def price_increase(price)
        @price*1.5
      end

    end

    class Magazine < Book

      def price_increase(price)
        @price*3
      end

    end
